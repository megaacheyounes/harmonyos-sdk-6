105  do { \
106  size_t __cpu = (cpu); \

133  do { \
134  cpu_set_t* __dst = (dstset); \
135  const __CPU_BITTYPE* __src1 = (srcset1)->__bits; \
136  const __CPU_BITTYPE* __src2 = (srcset2)->__bits; \
137  size_t __nn = 0, __nn_max = (setsize)/sizeof(__CPU_BITTYPE); \
